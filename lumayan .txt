local function safeVec(v)
if typeof(v) == "Vector3" then return v end
if typeof(v) == "table" and #v >= 3 then
return Vector3.new(tonumber(v[1]) or 0, tonumber(v[2]) or 0, tonumber(v[3]) or 0)
end
return Vector3.zero
end

local function playFrames(frames, onFinish)
if replayLock or #frames == 0 then return end
replayLock = true
if currentReplayToken then currentReplayToken.stop = true end

local token = { stop = false }
currentReplayToken = token
autoForward = true

local index = 1
local lastY = nil
local isJumping = false
local lastJumpTime = 0
local jumpCooldown = 0.5

while index < #frames and not token.stop and not stopAll do
if not rootPart or not humanoid or humanoid.Health <= 0 then
task.wait()
else
local i = math.floor(index)
local f, fNext = frames[i], frames[i + 1]
if f and fNext and f.Position and fNext.Position then
local pos1 = safeVec(f.Position)
local pos2 = safeVec(fNext.Position)
local look1 = safeVec(f.LookVector)
local look2 = safeVec(fNext.LookVector)
local up1 = safeVec(f.UpVector)
local up2 = safeVec(fNext.UpVector)
local alpha = index - i

local pos = pos1:Lerp(pos2, alpha)
local look = look1:Lerp(look2, alpha).Unit
local up = up1:Lerp(up2, alpha).Unit

-- MODE TURUN (Mini/Zepeto)
if modeTurun then
local currentOffset = (humanoid and humanoid.HipHeight or 2) + (rootPart and rootPart.Size.Y / 2 or 0)
local recordedOffset = (f.HipHeight or 2) + (f.RootSizeY or 0)
local offsetDiff = currentOffset - recordedOffset
pos = pos + Vector3.new(0, offsetDiff * 0.8 - 1, 0)
end

-- JUMP DETECTION
if lastY then
local deltaY = pos.Y - lastY
local now = tick()
if deltaY > 0.5 and not isJumping and (now - lastJumpTime) > jumpCooldown then
isJumping = true
lastJumpTime = now
pcall(function()
humanoid.Jump = true
humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
end)
end
if deltaY < -0.1 and isJumping then
isJumping = false
end
end

lastY = pos.Y

-- ‚úÖ MODE BACK DITERAPKAN DI SINI
pcall(function()
local finalLook = look
if _G.MODE_BACK then
finalLook = -look   -- üî• membalik 180¬∞
end

rootPart.CFrame = CFrame.lookAt(pos, pos + finalLook, up)
end)
end
end
index += teleportSpeed * 0.1
RunService.Heartbeat:Wait()
end

autoForward, replayLock, currentReplayToken = false, false, nil
if not stopAll and onFinish then onFinish() end
end

return
end

local targetPos = rootPart.Position
if lastReplayPos == nil then
lastReplayPos = targetPos
return
end

local delta = targetPos - lastReplayPos
local moveThreshold = 0.01

if delta.Magnitude > moveThreshold then
local fullMoveDir = delta.Unit
pcall(function()
humanoid:Move(fullMoveDir, false)
end)
else
pcall(function()
humanoid:Move(Vector3.new(0, 0, 0), false)
end)
end

lastReplayPos = targetPos
end)

CreateVIPButton({
    Title = "START",
    Icon = "circle-play",
    Color = Color3.fromRGB(80, 255, 120),
    Callback = function()

        if #savedReplays == 0 or not rootPart then return end

        -- üî• Gabungkan semua frame menjadi satu rute penuh
        local fullFrames = {}
        for _, replay in ipairs(savedReplays) do
            if replay.Frames then
                for _, f in ipairs(replay.Frames) do
                    table.insert(fullFrames, f)
                end
            end
        end

        if #fullFrames == 0 then return end

        -- üî• Cari frame terdekat dari seluruh rute (bukan checkpoint)
        local closestIndex, closestDist = 1, math.huge
        for i, f in ipairs(fullFrames) do
            if f.Position then
                local dist = (safeVec(f.Position) - rootPart.Position).Magnitude
                if dist < closestDist then
                    closestDist = dist
                    closestIndex = i
                end
            end
        end

        -- üî• Slice dari frame terdekat ‚Üí sampai FINISH
        local sliced = {}
        for i = closestIndex, #fullFrames do
            table.insert(sliced, fullFrames[i])
        end

        stopAll = false
        playFrames(sliced)
    end
})

if pages.IsFinished then

pages:AdvanceToNextPageAsync()
end
end
return servers

else

end
end
